// ---- Product Grid & Search/Filter ----
let currentCategory = "";
function showProducts(){
  const grid = document.getElementById('productsGrid');
  const searchVal = (document.getElementById('search')||{value:""}).value.toLowerCase();
  let displayData = productsData.filter(
    p => (p.name.toLowerCase().includes(searchVal) || p.category.toLowerCase().includes(searchVal) || 
          p.subcategory.toLowerCase().includes(searchVal) || p.desc.toLowerCase().includes(searchVal)) &&
         (currentCategory ? p.category === currentCategory : true)
  );
  grid.innerHTML = displayData.length ? "" : "<div style='padding:2em;color:#d94b39;'>No products found.</div>";
  displayData.forEach(product => {
    grid.innerHTML += `
      <div class="product-card" style="display:flex;flex-direction:column;">
        <img src="${product.img}" alt="${product.name}" style="width:100%;max-width:120px;margin:0 auto 1em auto;border-radius:9px;box-shadow:0 2px 8px #eee;" loading="lazy">
        <h4>${product.name}</h4>
        <div style="color:#111;"><b>${product.category}</b> — ${product.subcategory}</div>
        <div class="price">${product.price}</div>
        <div style="font-size:.98em;color:#555;margin:.2em 0 1em 0;">${product.specs}</div>
        <div style="margin-bottom:1em;color:#191919;">${product.desc}</div>
        <button class="btn" onclick="cartEnquiry('${product.name}')"><i class="fa fa-shopping-cart"></i> Buy / Enquire</button>
      </div>
    `;
  });
}
// Category Filter
function filterCategory(cat){
  currentCategory = cat;
  document.getElementById('search').value = '';
  showProducts();
  window.location.hash='#products';
}
// Cart/Enquiry Functionality
function cartEnquiry(productName){
  window.location.hash="#contact";
  setTimeout(function(){
    document.querySelector('select[name="interest"]').value = "General Enquiry";
    document.querySelector('textarea[name="message"]').value = "Enquiry for: " + productName;
    document.querySelector('input[name="name"]').focus();
  },150);
}
// Contact Success Function
function showContactSuccess(event){
  event.preventDefault();
  document.getElementById("contactSuccessMsg").style.display = "block";
  setTimeout(()=>{document.getElementById("contactSuccessMsg").style.display="none";}, 7000);
  event.target.reset();
  return false;
}
